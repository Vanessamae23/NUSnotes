class Box<T> {
    private final T content;
    private Box(T content) {
      this.content = content;
    }
    private static final Box<?> EMPTY_BOX = new Box<>(null);

    public static <T> Box<T> of(T x) {
      
      if (x == null) {
        return null;
      }

      return new Box<T>(x);
    }

    public static <T> Box<T> empty() {
      @SuppressWarnings("unchecked")
      Box<T> b = (Box<T>) EMPTY_BOX;
      return b;
    }

    public static <T> Box<T> ofNullable(T x) {
      if(x == null) {
        return empty();
      }

      return new Box<T>(x);
     }

    public Box<T> filter(BooleanCondition<? super T> result) {
      if(!isPresent()) {
        return empty();
      }
     
      if(result.test(this.content)) {
        return this;
      } else {
        return empty();
      }
    }

    public <T> boolean isPresent() {
      if(this == EMPTY_BOX) {
        return false;
      } else {
        return true;
      }
    }

    public <U> Box<U> map(Transformer<? super T, ? extends U> transformer) {
      if(!isPresent()) {
        return empty();
      }

      return Box.ofNullable(transformer.transform(this.content));
    }

    @Override
    public boolean equals(Object obj) {
      if(obj == this) {
          return true;
      }
       
      if (obj instanceof Box<?>) {
        Box<?> b = (Box<?>) obj;
        if(this.content == b.content) {
          return true;
        }

        if(this.content == null || b.content == null) {
          return false;
        }

        return this.content.equals(b.content);
      }
      return false;
      
    }

    @Override
    public String toString() {
      String str = "";
      if(this.content != null) {
        str = String.format("[%s]", this.content);
      } else {
        str = "[]";
      }
        
      return str;
    }
  }
